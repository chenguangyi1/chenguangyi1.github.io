MySQL基本操作

2020-12-12

添加数据

①添加数据

insert into 表名（字段名1，字段名2，。。。）values (‘ 值1’,’值2’,’。。’)

#example:

inset into users (id,username,password) values (‘18’,’keep’,’boom’);

②添加字段名

alter table 表名 add column 字段名 字段类型 default null (新增默认为空的字段名)

alter table 表名 add column 字段名 字段类型 not null (新增不为空的字段)

#example：

alter table email add column varchar(10) not null;

③添加表名

CREATE TABLE MyGuests (

id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,

firstname VARCHAR(30) NOT NULL,

lastname VARCHAR(30) NOT NULL,

email VARCHAR(50),

reg_date TIMESTAMP )

注：

NOT NULL - 每一行都必须含有值（不能为空），null 值是不允许的。

DEFAULT value - 设置默认值。

UNSIGNED - 使用无符号数值类型，0 及正数 。

AUTO INCREMENT - 设置 MySQL 字段的值在新增记录时每次自动增长 1。

PRIMARY KEY - 设置数据表中每条记录的唯一标识。 通常列的 PRIMARY KEY 设置为 ID 数值，与 AUTO_INCREMENT 一起使用。

#example:

create table email1 like email;

④添加数据库

create database 数据库名

#exmaple:

create database keepboom;

删除数据

①删除数据

delete from 表名 where(可选) 条件

#example：

delete from email where id=1;

删除全部数据：delete from email;

②删除字段名

alter table 表名 drop column 字段名

#exmaple:

alter table email drop column id;

③删除表名

drop table 表名

#example：

drop table email;

④删除数据库

drop database 数据库名

#example：

drop database security；

修改数据

①修改数据

update 表名 set 字段名1=‘值1’，字段名2=‘值2’ where 条件

#exmaple：

update email set id_email=’chen’ where id=10;

②修改字段名

alter table 表名 change 旧字段 新字段 类型

#example：

alter table email change id id2 varchar(10);

③修改表名

rename table 旧表名 to 新表名

#example:

rename table email to email1;

alter table 旧表名 rename to 新表名

#example：

alter table email rename to email1;

查找数据

①查找数据

select 想查的 from 表名 where 条件

#example：

select username,password from users where id=1

②查找字段名

desc 表名

#example:

desc users;

③查找表名

show tables；

④查找数据库

show databases；
